const resultDiv = document.getElementById('js-result'); // 入力フィールドとボタンを作成 const input1 = document.createElement('input'); input1.type = 'number'; const input2 = document.createElement('input'); input2.type = 'number'; const addButton = document.createElement('button'); addButton.textContent = '足し算'; const subtractButton = document.createElement('button'); subtractButton.textContent = '引き算'; const result = document.createElement('p'); // 要素を追加 resultDiv.appendChild(input1); resultDiv.appendChild(input2); resultDiv.appendChild(addButton); resultDiv.appendChild(subtractButton); resultDiv.appendChild(result); // 足し算の関数 function add() { const num1 = parseFloat(input1.value); const num2 = parseFloat(input2.value); const sum = num1 + num2; result.textContent = 結果: ${sum}; } // 引き算の関数 function subtract() { const num1 = parseFloat(input1.value); const num2 = parseFloat(input2.value); const difference = num1 - num2; result.textContent = 結果: ${difference}; } // ボタンにイベントリスナーを追加 addButton.addEventListener('click', add); subtractButton.addEventListener('click', subtract);
